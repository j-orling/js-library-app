CREATE TABLE `personal_library`.`books` (`book_id` INT NOT NULL AUTO_INCREMENT , `book_title` VARCHAR(50) NOT NULL , `book_author` VARCHAR(50) NOT NULL , `book_read` VARCHAR(10) NOT NULL DEFAULT 'NOT READ' , `book_readDate` DATE NULL DEFAULT NULL , `book_rating` INT NULL DEFAULT NULL , PRIMARY KEY (`book_id`)) ENGINE = InnoDB;
